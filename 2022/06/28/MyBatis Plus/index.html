1.1 简介

MyBatis-Plus（简称 MP）是一个 MyBatis的增强工具，在 MyBatis 的基础上只做增强不做改变，为简化开发、提高效率而生

2.2 NULL值处理

@Test
void testNull(){
    UserQuery uq = new UserQuery();
    uq.setAge(10);
    uq.setAge2(30);

    LambdaQueryWrapper<User> lqw = new LambdaQueryWrapper();
    lqw.lt(uq.getAge()!=null,User::getAge,uq.getAge())
        .gt(uq.getAge2()!=null,User::getAge,uq.getAge2());
    List<User> users = userDao.selectList(lqw);
    System.out.println(users);
}

2.3 查询投影

  • 查询出来看到的结果

@Test
void testProject(){
    QueryWrapper<User> qw = new QueryWrapper<User>();
    qw.select("count(*) as count");
    List<User> users = userDao.selectList(qw);
    System.out.println(users);
}

//查询结果是封装到了User对象中，实体类中没有count这个字段与之对应，所以结果为null


@Test
void testProject(){
    QueryWrapper<User> qw = new QueryWrapper<User>();
    qw.select("count(*) as count");
    List<Map<String, Object>> maps = userDao.selectMaps(qw);
    System.out.println(maps);
}


@Test
void testProject(){
    QueryWrapper<User> qw = new QueryWrapper<User>();
    qw.select("count(*) as count,age");
    qw.groupBy("age");
    List<Map<String, Object>> maps = userDao.selectMaps(qw);
    System.out.println(maps);
}

2.10 代码生成器

  1. 导入相关依赖

    <!--代码生成器-->
<dependency>
    <groupId>com.baomidou</groupId>
    <artifactId>mybatis-plus-generator</artifactId>
    <version>3.0.2</version>
</dependency>
<!--velocity模板引擎-->
<dependency>
    <groupId>org.apache.velocity</groupId>
    <artifactId>velocity-engine-core</artifactId>
    <version>2.0</version>
</dependency>

  2. 创建代码生成器对象，执行代码生成操作

    //创建代码生成器对象
AutoGenerator ag = new AutoGenerator();
//数据源配置,读取数据库中的信息,根据数据库表结构生成代码
DataSourceConfig datasource = new DataSourceConfig();
datasource.setDriverName("com.mysql.cj.jdbc.Driver");
datasource.setUrl("jdbc:mysql://localhost:3306/mybatisplus?useUnicode=true&characterEncoding=UTF-8&userSSL=false&serverTimezone=GMT%2B8");
datasource.setUsername("root");
datasource.setPassword("123456");
ag.setDataSource(datasource);

//全局配置
GlobalConfig globalConfig = new GlobalConfig();
globalConfig.setAuthor("杨奇");
globalConfig.setOutputDir("C:\\Users\\杨奇\\Desktop");
globalConfig.setFileOverride(true);
globalConfig.setMapperName("%sDao");
globalConfig.setIdType(IdType.ID_WORKER);
ag.setGlobalConfig(globalConfig);

//包相关配置
PackageConfig packageConfig = new PackageConfig();
packageConfig.setParent("com.yunmo");
packageConfig.setController("controller");
packageConfig.setEntity("pojo");
packageConfig.setMapper("dao");
ag.setPackageInfo(packageConfig);

//策略配置
StrategyConfig strategyConfig = new StrategyConfig();
strategyConfig.setInclude("t_user");
strategyConfig.setLogicDeleteFieldName("deleted");
strategyConfig.setVersionFieldName("version");
strategyConfig.setTablePrefix("t_");
strategyConfig.setRestControllerStyle(true);
strategyConfig.setEntityLombokModel(true);
ag.setStrategy(strategyConfig);

ag.execute();
